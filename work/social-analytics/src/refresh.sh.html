#!/bin/bash
# Weekly refresh — run every Sunday after dropping new CSV exports into ref/data/
# Usage: bash ref/scripts/refresh.sh
set -e

cd "$(dirname "$0")/../.."

echo "[$(date '+%Y-%m-%d %H:%M')] Building workbooks…"
python3 ref/scripts/build_workbooks.py
echo "[$(date '+%Y-%m-%d %H:%M')] Building dashboards…"
python3 ref/scripts/build_dashboards.py
echo "[$(date '+%Y-%m-%d %H:%M')] Done."
