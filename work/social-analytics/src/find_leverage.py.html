"""One-shot analysis: surface the highest-leverage findings from current social data.

Outputs a structured findings doc to stdout. Captures calculations Michael (Chief of Staff,
owns IG personally, oversees rest) would use to decide where to put effort.
"""
from __future__ import annotations
import warnings
from pathlib import Path
import pandas as pd

warnings.filterwarnings("ignore")
BASE = Path("/Users/dags/Desktop/PubKey/PubKey-Ops/Social")


def _glob_concat(folder: Path, pattern: str, date_col="Date", date_fmt="%m-%d-%Y") -> pd.DataFrame:
    files = sorted(folder.glob(pattern))
    df = pd.concat([pd.read_csv(f) for f in files], ignore_index=True)
    if date_col in df.columns:
        df[date_col] = pd.to_datetime(df[date_col], format=date_fmt, errors="coerce")
        keys = [c for c in [date_col, "Network", "Profile"] if c in df.columns]
        df = df.drop_duplicates(subset=keys, keep="last").sort_values(date_col).reset_index(drop=True)
    return df


def _to_num(s: pd.Series) -> pd.Series:
    if s.dtype == object:
        s = s.astype(str).str.replace(",", "", regex=False).replace({"": None, "nan": None})
    return pd.to_numeric(s, errors="coerce")


def load_overview() -> pd.DataFrame:
    df = _glob_concat(BASE / "ref/data/overview/source", "Profile Performance*.csv")
    for c in ["Audience", "Net Audience Growth", "Published Posts (Total)",
              "Impressions", "Engagements", "Video Views"]:
        if c in df.columns:
            df[c] = _to_num(df[c])
    return df


def per_network_efficiency(ov: pd.DataFrame) -> pd.DataFrame:
    """Engagements per post, ER per impression, audience growth rate per day."""
    last_90 = ov[ov.Date >= ov.Date.max() - pd.Timedelta(days=90)]
    rows = []
    for network, g in last_90.groupby("Network"):
        posts = g["Published Posts (Total)"].sum()
        impr = g["Impressions"].sum()
        eng = g["Engagements"].sum()
        net_growth = g["Net Audience Growth"].sum()
        latest_audience = g.sort_values("Date").iloc[-1]["Audience"] if not g.empty else None
        days_active = (g["Published Posts (Total)"] > 0).sum()
        rows.append({
            "Network": network,
            "Audience": int(latest_audience) if pd.notna(latest_audience) else None,
            "Posts/90d": int(posts) if pd.notna(posts) else 0,
            "Posting days/90": int(days_active),
            "Impressions/90d": int(impr) if pd.notna(impr) else 0,
            "Engagements/90d": int(eng) if pd.notna(eng) else 0,
            "Net new followers/90d": int(net_growth) if pd.notna(net_growth) else 0,
            "Eng/post": round(eng / posts, 1) if posts and pd.notna(eng) else 0,
            "ER/impression": round(eng / impr * 100, 2) if impr and pd.notna(eng) else 0,
            "Impr/post": round(impr / posts, 0) if posts and pd.notna(impr) else 0,
            "Followers/post": round(net_growth / posts, 2) if posts and pd.notna(net_growth) else 0,
        })
    out = pd.DataFrame(rows).sort_values("Audience", ascending=False)
    return out


def youtube_video_distribution() -> pd.DataFrame:
    folder = BASE / "ref/data/youtube/source"
    files = sorted(folder.glob("YouTube Videos*.csv"))
    df = pd.concat([pd.read_csv(f) for f in files], ignore_index=True)
    df = df.drop_duplicates(subset=["Video ID"], keep="last")
    for c in ["Video Views", "Estimated Minutes Watched (Minutes)",
              "Average View Duration (Seconds)", "Average View Percentage",
              "Engagements", "Likes", "Comments", "Subscribers Gained from Video"]:
        if c in df.columns:
            df[c] = _to_num(df[c])
    df = df[df["Video Views"].notna() & (df["Video Views"] > 0)]
    return df


def youtube_concentration(yt: pd.DataFrame) -> dict:
    """How much of YouTube outcomes are driven by top % of videos?"""
    yt_sorted = yt.sort_values("Video Views", ascending=False)
    total_views = yt["Video Views"].sum()
    total_subs = yt["Subscribers Gained from Video"].sum()
    top_10 = yt_sorted.head(int(len(yt) * 0.10))
    top_25 = yt_sorted.head(int(len(yt) * 0.25))
    return {
        "Total videos": len(yt),
        "Total views (all)": int(total_views),
        "Total subs gained (all)": int(total_subs) if pd.notna(total_subs) else 0,
        "Top 10% video count": len(top_10),
        "Top 10% share of views": f"{top_10['Video Views'].sum() / total_views * 100:.1f}%",
        "Top 10% share of subs": f"{top_10['Subscribers Gained from Video'].sum() / total_subs * 100:.1f}%" if total_subs else "n/a",
        "Top 25% share of views": f"{top_25['Video Views'].sum() / total_views * 100:.1f}%",
        "Median video views": int(yt["Video Views"].median()),
        "Mean video views": int(yt["Video Views"].mean()),
        "Median view-through %": round(yt["Average View Percentage"].median(), 1),
        "Videos with 0 subs gained": int((yt["Subscribers Gained from Video"] == 0).sum()),
        "Pct videos with 0 subs": f"{(yt['Subscribers Gained from Video'] == 0).sum() / len(yt) * 100:.1f}%",
    }


def linkedin_post_efficiency() -> dict:
    """LinkedIn-specific deep dive — high ER per impression but tiny absolute volume."""
    f = sorted((BASE / "ref/data/linkedin/source").glob("LinkedIn Pages*.csv"))[0]
    df = pd.read_csv(f)
    df["Date"] = pd.to_datetime(df["Date"], format="%m-%d-%Y", errors="coerce")
    for c in ["Followers", "Impressions", "Engagements", "Published Posts"]:
        df[c] = _to_num(df[c])
    posting_days = df[df["Published Posts"] > 0]
    last_90 = df[df.Date >= df.Date.max() - pd.Timedelta(days=90)]
    p90 = last_90[last_90["Published Posts"] > 0]
    return {
        "Total posting days (440d)": len(posting_days),
        "Posting cadence": f"{len(posting_days) / 440 * 7:.1f} posts/week",
        "Avg impressions/posting day": int(posting_days["Impressions"].mean()),
        "Avg engagements/posting day": int(posting_days["Engagements"].mean()),
        "ER on posting days (lifetime)": f"{(posting_days['Engagements'] / posting_days['Impressions']).mean() * 100:.1f}%",
        "Net follower growth (440d)": int(df.iloc[-1]["Followers"] - df["Followers"].dropna().iloc[0]),
        "Followers per posting day": round((df.iloc[-1]["Followers"] - df["Followers"].dropna().iloc[0]) / len(posting_days), 2),
        "Last 90d posting days": len(p90),
        "Last 90d cadence": f"{len(p90) / 90 * 7:.1f} posts/week",
    }


def instagram_view_dropoff() -> dict:
    f = sorted((BASE / "ref/data/instagram/source").glob("Instagram Business Profiles*.csv"))[0]
    df = pd.read_csv(f)
    df["Date"] = pd.to_datetime(df["Date"], format="%m-%d-%Y", errors="coerce")
    for c in ["Followers", "Views", "Reach", "Organic Engagements", "Published Posts"]:
        df[c] = _to_num(df[c])
    posting_days = df[df["Published Posts"] > 0]
    return {
        "Days of data": len(df),
        "Posting days": len(posting_days),
        "Posts in window": int(df["Published Posts"].sum()),
        "Posting cadence": f"{df['Published Posts'].sum() / len(df) * 7:.1f} posts/week",
        "Total views": int(df["Views"].sum()),
        "Total reach": int(df["Reach"].sum()),
        "Views/post avg": int(df["Views"].sum() / df["Published Posts"].sum()) if df["Published Posts"].sum() else 0,
        "Reach/post avg": int(df["Reach"].sum() / df["Published Posts"].sum()) if df["Published Posts"].sum() else 0,
        "View-to-reach ratio": f"{df['Views'].sum() / df['Reach'].sum():.2f}x" if df["Reach"].sum() else "n/a",
        "Follower delta (window)": int(df.iloc[-1]["Followers"] - df["Followers"].dropna().iloc[0]),
        "Followers per post": round((df.iloc[-1]["Followers"] - df["Followers"].dropna().iloc[0]) / df["Published Posts"].sum(), 2) if df["Published Posts"].sum() else 0,
    }


def cross_channel_effort_vs_outcome(by_network: pd.DataFrame) -> str:
    """Build the 'where should next 90 days of effort go' table."""
    # Rank by composite leverage = engagements/post weighted by audience reach potential
    df = by_network.copy()
    df["audience_pct"] = df["Audience"] / df["Audience"].sum() * 100 if df["Audience"].sum() else 0
    df["effort_share"] = df["Posts/90d"] / df["Posts/90d"].sum() * 100 if df["Posts/90d"].sum() else 0
    df["growth_share"] = df["Net new followers/90d"].abs() / df["Net new followers/90d"].abs().sum() * 100 if df["Net new followers/90d"].abs().sum() else 0
    df["effort_to_growth_gap"] = df["growth_share"] - df["effort_share"]
    return df[["Network", "Audience", "Posts/90d", "Net new followers/90d",
               "effort_share", "growth_share", "effort_to_growth_gap"]].sort_values("effort_to_growth_gap", ascending=False).to_string(index=False)


def main():
    print("=" * 80)
    print("PUBKEY SOCIAL · LEVERAGE FINDINGS · 2026-05-15")
    print("=" * 80)
    print()

    ov = load_overview()
    print(f"Cross-channel coverage: {ov.Date.min().date()} → {ov.Date.max().date()} · {ov.Network.nunique()} networks · {len(ov):,} rows")
    print()

    print("─" * 80)
    print("FINDING 1 · PER-NETWORK 90-DAY EFFICIENCY")
    print("─" * 80)
    eff = per_network_efficiency(ov)
    print(eff.to_string(index=False))
    print()

    print("─" * 80)
    print("FINDING 2 · EFFORT vs OUTCOME GAP (90d)")
    print("─" * 80)
    print("effort_share = % of total posts on this network")
    print("growth_share = % of total new followers from this network")
    print("gap = where to reallocate (positive = under-invested, negative = over-invested)")
    print()
    print(cross_channel_effort_vs_outcome(eff))
    print()

    print("─" * 80)
    print("FINDING 3 · YOUTUBE OUTCOME CONCENTRATION")
    print("─" * 80)
    yt = youtube_video_distribution()
    for k, v in youtube_concentration(yt).items():
        print(f"  {k:<35} {v}")
    print()

    print("─" * 80)
    print("FINDING 4 · LINKEDIN DEEP DIVE")
    print("─" * 80)
    for k, v in linkedin_post_efficiency().items():
        print(f"  {k:<35} {v}")
    print()

    print("─" * 80)
    print("FINDING 5 · INSTAGRAM DEEP DIVE (75d window)")
    print("─" * 80)
    for k, v in instagram_view_dropoff().items():
        print(f"  {k:<35} {v}")
    print()


if __name__ == "__main__":
    main()
