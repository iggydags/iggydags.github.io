def _glob_concat(folder, pattern, date_col="Date", date_format="%m-%d-%Y"):
    """Glob, concat, dedupe by date, sort."""
    files = sorted(folder.glob(pattern))
    if not files:
        raise FileNotFoundError(f"No files matching {pattern} in {folder}")
    df = pd.concat([pd.read_csv(f) for f in files], ignore_index=True)
    df[date_col] = pd.to_datetime(df[date_col], format=date_format, errors="coerce")
    return df.drop_duplicates(subset=[date_col], keep="last").sort_values(date_col).reset_index(drop=True)

def kpi(label: str, num: str, sub: str = "", variant: str = "", windows: dict = None) -> str:

    cls = f"kpi {variant}".strip()

  <div class="label">{escape(label)}</div>
  {num_el}
  <div class="sub">{escape(sub) if "<" not in sub else sub}</div>
</div>"""

def line_chart(points: list[tuple[float, float]], width: int = 600, height: int = 240,
               y_label: str = "", x_labels: list[str] = None, fill: bool = True,
               annotations: list[tuple[int, str]] = None) -> str:
    """points = list of (x, y) tuples; x is numeric. Returns SVG string."""
    if not points:
        return f'<svg class="chart" viewBox="0 0 {width} {height}"></svg>'
    pad_l, pad_r, pad_t, pad_b = 48, 16, 16, 32
    plot_w = width - pad_l - pad_r
    plot_h = height - pad_t - pad_b

    xs = [p[0] for p in points]
    ys = [p[1] for p in points]
    x_min, x_max = min(xs), max(xs)
    y_min, y_max = 0, max(ys) * 1.05 or 1
    x_range = (x_max - x_min) or 1
    y_range = (y_max - y_min) or 1

    def sx(x): return pad_l + (x - x_min) / x_range * plot_w
    def sy(y): return pad_t + plot_h - (y - y_min) / y_range * plot_h

    # gridlines (4 horizontal)
    grid = ""
    y_ticks = []
    for i in range(5):
        y_val = y_min + (y_range * i / 4)
        y_pos = sy(y_val)
        grid += f'<line x1="{pad_l}" y1="{y_pos:.1f}" x2="{pad_l+plot_w}" y2="{y_pos:.1f}"/>'
        y_ticks.append(f'<text x="{pad_l-6}" y="{y_pos+3:.1f}" text-anchor="end">{int(y_val):,}</text>')

    # path
    pts_str = " ".join(f"{sx(x):.1f},{sy(y):.1f}" for x, y in points)
    area_str = ""
    if fill:
        area_path = f"M {sx(x_min):.1f},{sy(y_min):.1f} L " + pts_str.replace(",", " ", 1) + f" L {sx(x_max):.1f},{sy(y_min):.1f} Z"
        area_str = f'<path class="area" d="{area_path}"/>'

    # x labels (5 ticks)
    x_label_svg = ""
    if x_labels:
        n = len(x_labels)
        for i, lab in enumerate(x_labels):
            xpos = pad_l + (i / (n - 1 or 1)) * plot_w
            x_label_svg += f'<text class="label" x="{xpos:.1f}" y="{height-10}" text-anchor="middle">{escape(lab)}</text>'

    ann_svg = ""
    for idx, text in (annotations or []):
        if not (0 <= idx < len(points)):
            continue
        x, y = points[idx]
        cx, cy = sx(x), sy(y)
        text_y = max(cy - 22, 14)
        anchor = "middle"
        if cx < 80:
            anchor = "start"
        elif cx > width - 80:
            anchor = "end"
        ann_svg += (
            f'<g class="annotation">'
            f'<line class="ann-line" x1="{cx:.1f}" y1="{text_y+6:.1f}" x2="{cx:.1f}" y2="{cy-3:.1f}"/>'
            f'<circle class="ann-dot" cx="{cx:.1f}" cy="{cy:.1f}" r="4"/>'
            f'<text class="ann-text" x="{cx:.1f}" y="{text_y:.1f}" text-anchor="{anchor}">{escape(text)}</text>'
            f'</g>'
        )

    return f'''<svg class="chart" viewBox="0 0 {width} {height}" preserveAspectRatio="none" role="img" aria-label="{escape(y_label)} line chart">
<g class="grid">{grid}</g>
{area_str}
<polyline class="line" points="{pts_str}"/>
<g class="axis">{"".join(y_ticks)}</g>
<g>{x_label_svg}</g>
{ann_svg}
</svg>'''


def bar_chart(items: list[tuple[str, float]], width: int = 600, height: int = 240,
              max_bars: int = 30, color: str = "orange",
              annotations: list[tuple[int, str]] = None) -> str:
    """items = list of (label, value). Vertical bars.
    annotations: list of (item_index, callout_text). Renders an orange leader + DM Mono callout above the bar."""
    if not items:
        return f'<svg class="chart" viewBox="0 0 {width} {height}"></svg>'
    items = items[:max_bars]
    pad_l, pad_r, pad_t, pad_b = 48, 16, 36, 36
    plot_w = width - pad_l - pad_r
    plot_h = height - pad_t - pad_b

    n = len(items)
    bar_w = plot_w / n * 0.7
    gap = plot_w / n * 0.3
    y_max = max(v for _, v in items) * 1.05 or 1

    grid = ""
    y_ticks = []
    for i in range(5):
        y_val = y_max * i / 4
        y_pos = pad_t + plot_h - (y_val / y_max) * plot_h
        grid += f'<line x1="{pad_l}" y1="{y_pos:.1f}" x2="{pad_l+plot_w}" y2="{y_pos:.1f}"/>'
        y_ticks.append(f'<text x="{pad_l-6}" y="{y_pos+3:.1f}" text-anchor="end">{int(y_val):,}</text>')

    bars = ""
    labels = ""
    cls = "bar" if color == "orange" else "bar alt"
    bar_centers = {}
    bar_tops = {}
    for i, (lab, v) in enumerate(items):
        x = pad_l + i * (bar_w + gap) + gap / 2
        bar_h = (v / y_max) * plot_h
        y = pad_t + plot_h - bar_h
        bars += f'<rect class="{cls}" x="{x:.1f}" y="{y:.1f}" width="{bar_w:.1f}" height="{bar_h:.1f}"/>'
        bar_centers[i] = x + bar_w / 2
        bar_tops[i] = y
        if n <= 12 or i % max(1, n // 8) == 0:
            label_x = x + bar_w / 2
            labels += f'<text class="label" x="{label_x:.1f}" y="{height-10}" text-anchor="middle">{escape(str(lab))}</text>'

    ann_svg = ""
    for idx, text in (annotations or []):
        if idx not in bar_centers:
            continue
        cx = bar_centers[idx]
        top = bar_tops[idx]
        text_y = max(top - 22, 14)
        # Anchor text — centered on bar, but clamp to viewport
        anchor = "middle"
        if cx < 80:
            anchor = "start"
        elif cx > width - 80:
            anchor = "end"
        ann_svg += (
            f'<g class="annotation">'
            f'<line class="ann-line" x1="{cx:.1f}" y1="{text_y+6:.1f}" x2="{cx:.1f}" y2="{top-2:.1f}"/>'
            f'<circle class="ann-dot" cx="{cx:.1f}" cy="{top:.1f}" r="4"/>'
            f'<text class="ann-text" x="{cx:.1f}" y="{text_y:.1f}" text-anchor="{anchor}">{escape(text)}</text>'
            f'</g>'
        )

    return f'''<svg class="chart" viewBox="0 0 {width} {height}" preserveAspectRatio="none" role="img">
<g class="grid">{grid}</g>
{bars}
<g class="axis">{"".join(y_ticks)}</g>
<g>{labels}</g>
{ann_svg}
</svg>'''


def hbar_chart(items: list[tuple[str, float]], width: int = 480, height: int = 280,
               max_bars: int = 8) -> str:
    """items = list of (label, value). Horizontal bars."""
    if not items:
        return f'<svg class="chart" viewBox="0 0 {width} {height}"></svg>'
    items = items[:max_bars]
    pad_l, pad_r, pad_t, pad_b = 140, 60, 8, 8
    plot_w = width - pad_l - pad_r
    plot_h = height - pad_t - pad_b

    n = len(items)
    bar_h = plot_h / n * 0.7
    gap = plot_h / n * 0.3
    v_max = max(v for _, v in items) or 1

    bars = ""
    for i, (lab, v) in enumerate(items):
        y = pad_t + i * (bar_h + gap) + gap / 2
        w = (v / v_max) * plot_w
        bars += f'<rect class="bar" x="{pad_l}" y="{y:.1f}" width="{w:.1f}" height="{bar_h:.1f}"/>'
        bars += f'<text class="label" x="{pad_l-8}" y="{y+bar_h*0.65:.1f}" text-anchor="end">{escape(str(lab)[:24])}</text>'
        bars += f'<text class="label" x="{pad_l+w+6:.1f}" y="{y+bar_h*0.65:.1f}" style="fill:var(--paper);font-weight:600">{int(v):,}</text>'

    return f'''<svg class="chart" viewBox="0 0 {width} {height}" preserveAspectRatio="xMidYMid meet" role="img">
{bars}
</svg>'''


def donut_chart(items: list[tuple[str, float]], width: int = 280, height: int = 180,
                center_label: str = "", center_value: str = "") -> str:
    """Donut chart for composition. items = list of (label, value)."""
    if not items:
        return f'<svg class="chart" viewBox="0 0 {width} {height}"></svg>'
    total = sum(v for _, v in items) or 1
    cx, cy = width / 2 - 60, height / 2
    r_outer, r_inner = 70, 44
    palette = ["var(--orange)", "var(--blue)", "var(--amber)", "var(--green)", "var(--orange-hot)", "var(--paper-mute)"]

    import math
    angle_start = -math.pi / 2  # start at 12 o'clock
    arcs = ""
    legend_html = ""
    for i, (label, value) in enumerate(items):
        frac = value / total
        if frac <= 0:
            continue
        angle_end = angle_start + frac * 2 * math.pi
        large = 1 if frac > 0.5 else 0
        x1, y1 = cx + r_outer * math.cos(angle_start), cy + r_outer * math.sin(angle_start)
        x2, y2 = cx + r_outer * math.cos(angle_end), cy + r_outer * math.sin(angle_end)
        x3, y3 = cx + r_inner * math.cos(angle_end), cy + r_inner * math.sin(angle_end)
        x4, y4 = cx + r_inner * math.cos(angle_start), cy + r_inner * math.sin(angle_start)
        path = (f"M {x1:.1f},{y1:.1f} A {r_outer},{r_outer} 0 {large} 1 {x2:.1f},{y2:.1f} "
                f"L {x3:.1f},{y3:.1f} A {r_inner},{r_inner} 0 {large} 0 {x4:.1f},{y4:.1f} Z")
        color = palette[i % len(palette)]
        arcs += f'<path d="{path}" fill="{color}" stroke="var(--ink-2)" stroke-width="1.5"/>'
        pct = frac * 100
        legend_html += (
            f'<g class="leg"><rect x="{width-130}" y="{20+i*22}" width="10" height="10" fill="{color}"/>'
            f'<text x="{width-114}" y="{29+i*22}" font-family="DM Mono" font-size="10" fill="var(--paper-dim)">'
            f'{escape(str(label)[:18])}</text>'
            f'<text x="{width-6}" y="{29+i*22}" font-family="DM Mono" font-size="10" font-weight="600" '
            f'fill="var(--paper)" text-anchor="end">{pct:.0f}%</text></g>'
        )
        angle_start = angle_end

    center = ""
    if center_label or center_value:
        center = (
            f'<text x="{cx}" y="{cy-2}" text-anchor="middle" font-family="Barlow Condensed" font-weight="800" '
            f'font-size="22" fill="var(--paper)" letter-spacing="-0.02em">{escape(center_value)}</text>'
            f'<text x="{cx}" y="{cy+14}" text-anchor="middle" font-family="DM Mono" font-size="9" '
            f'letter-spacing="0.16em" fill="var(--paper-mute)">{escape(center_label.upper())}</text>'
        )
    return f'<svg class="chart" viewBox="0 0 {width} {height}" preserveAspectRatio="xMidYMid meet" role="img">{arcs}{center}{legend_html}</svg>'


def stacked_bar_chart(items: list[tuple[str, float]], total_label: str = "",
                     width: int = 320, height: int = 60) -> str:
    """Single horizontal stacked bar showing composition. items = (label, value)."""
    if not items:
        return ''
    total = sum(v for _, v in items) or 1
    palette = ["var(--orange)", "var(--blue)", "var(--amber)", "var(--green)", "var(--orange-hot)", "var(--paper-mute)"]
    bar_h, bar_y = 22, 8
    bar_w = width - 4
    x = 2
    segs = ""
    legend = ""
    for i, (label, value) in enumerate(items):
        frac = value / total
        seg_w = frac * bar_w
        color = palette[i % len(palette)]
        segs += f'<rect x="{x:.1f}" y="{bar_y}" width="{seg_w:.1f}" height="{bar_h}" fill="{color}"/>'
        # Label below if segment wide enough
        if seg_w > 35:
            tx = x + seg_w / 2
            segs += (f'<text x="{tx:.1f}" y="{bar_y + bar_h/2 + 4}" text-anchor="middle" '
                     f'font-family="DM Mono" font-size="10" font-weight="600" fill="white">{frac*100:.0f}%</text>')
        legend += (f'<g><rect x="{x:.1f}" y="{bar_y + bar_h + 8}" width="8" height="8" fill="{color}"/>'
                   f'<text x="{x+12:.1f}" y="{bar_y + bar_h + 16}" font-family="DM Mono" font-size="9" '
                   f'fill="var(--paper-dim)">{escape(str(label)[:14])}</text></g>')
        x += seg_w
    return f'<svg class="chart" viewBox="0 0 {width} {height}" preserveAspectRatio="xMidYMid meet" role="img">{segs}{legend}</svg>'


def pct_bars(items: list[tuple[str, float, float]], width: int = 320, height: int = 200,
             max_pct: float = None) -> str:
    """Horizontal bars with percentage values. items = (label, value, max_value).
    The bar fills based on value/max_value * width; pct shown at end."""
    if not items:
        return ''
    pad_l, pad_r = 100, 50
    plot_w = width - pad_l - pad_r
    n = len(items)
    bar_h = (height - 16) / n * 0.7
    gap = (height - 16) / n * 0.3
    elements = ""
    for i, (label, value, m) in enumerate(items):
        y = 8 + i * (bar_h + gap) + gap / 2
        pct = (value / m) if m else 0
        w = pct * plot_w
        elements += f'<text x="{pad_l-8:.1f}" y="{y+bar_h*0.65:.1f}" text-anchor="end" font-family="DM Mono" font-size="10" fill="var(--paper-dim)" letter-spacing="0.04em">{escape(str(label)[:18])}</text>'
        elements += f'<rect x="{pad_l}" y="{y:.1f}" width="{plot_w:.1f}" height="{bar_h:.1f}" fill="var(--line-soft)"/>'
        elements += f'<rect x="{pad_l}" y="{y:.1f}" width="{w:.1f}" height="{bar_h:.1f}" fill="var(--orange)"/>'
        elements += f'<text x="{pad_l+plot_w+6:.1f}" y="{y+bar_h*0.65:.1f}" font-family="DM Mono" font-size="10" font-weight="600" fill="var(--paper)">{pct*100:.1f}%</text>'
    return f'<svg class="chart" viewBox="0 0 {width} {height}" preserveAspectRatio="xMidYMid meet" role="img">{elements}</svg>'


def sparkline(values: list[float], width: int = 110, height: int = 26, color: str = "orange") -> str:
    """Tiny inline line chart for KPI trends. No axes, just shape."""
    vals = [v for v in values if v is not None and not (isinstance(v, float) and pd.isna(v))]
    if len(vals) < 2:
        return ''
    v_max, v_min = max(vals), min(vals)
    span = v_max - v_min or 1
    n = len(vals)
    stroke = "var(--orange-hot)" if color == "orange" else f"var(--{color})"
    pts = ' '.join(
        f'{i*(width-2)/(n-1)+1:.1f},{height - 2 - (v - v_min)/span * (height-4):.1f}'
        for i, v in enumerate(vals)
    )
    # area path for soft fill underneath
    area = f'M 1,{height-2} L ' + pts.replace(',', ' ', 1) + f' L {width-1},{height-2} Z'
    return (
        f'<svg class="spark" viewBox="0 0 {width} {height}" preserveAspectRatio="none" '
        f'width="{width}" height="{height}" aria-hidden="true">'
        f'<path d="{area}" fill="{stroke}" opacity="0.12"/>'
        f'<polyline points="{pts}" fill="none" stroke="{stroke}" stroke-width="1.4" stroke-linecap="round" stroke-linejoin="round"/>'
        f'</svg>'
    )


def scatter_chart(points: list[tuple[float, float, str]], width: int = 600, height: int = 280,
                  x_label: str = "", y_label: str = "",
                  annotations: list[tuple[int, str]] = None) -> str:
    """points = list of (x, y, label). annotations: list of (point_index, callout_text)."""
    if not points:
        return f'<svg class="chart" viewBox="0 0 {width} {height}"></svg>'
    pad_l, pad_r, pad_t, pad_b = 56, 16, 16, 36
    plot_w = width - pad_l - pad_r
    plot_h = height - pad_t - pad_b

    xs = [p[0] for p in points]
    ys = [p[1] for p in points]
    x_max = max(xs) * 1.05 or 1
    y_max = max(ys) * 1.05 or 1

    def sx(x): return pad_l + (x / x_max) * plot_w
    def sy(y): return pad_t + plot_h - (y / y_max) * plot_h

    grid = ""
    y_ticks = []
    x_ticks = []
    for i in range(5):
        # horizontal grid
        y_val = y_max * i / 4
        y_pos = sy(y_val)
        grid += f'<line x1="{pad_l}" y1="{y_pos:.1f}" x2="{pad_l+plot_w}" y2="{y_pos:.1f}"/>'
        y_ticks.append(f'<text x="{pad_l-6}" y="{y_pos+3:.1f}" text-anchor="end">{int(y_val):,}</text>')
        # x ticks
        x_val = x_max * i / 4
        x_pos = sx(x_val)
        x_ticks.append(f'<text class="label" x="{x_pos:.1f}" y="{height-10}" text-anchor="middle">{int(x_val):,}</text>')

    dots = ""
    for x, y, _ in points:
        dots += f'<circle class="dot" cx="{sx(x):.1f}" cy="{sy(y):.1f}" r="3" opacity="0.6"/>'

    ann_svg = ""
    for idx, text in (annotations or []):
        if not (0 <= idx < len(points)):
            continue
        x, y, _ = points[idx]
        cx, cy = sx(x), sy(y)
        # Place text to right of point if there's room, else left
        if cx < width - 140:
            text_x, anchor = cx + 12, "start"
        else:
            text_x, anchor = cx - 12, "end"
        ann_svg += (
            f'<g class="annotation">'
            f'<line class="ann-line" x1="{cx:.1f}" y1="{cy:.1f}" x2="{text_x:.1f}" y2="{cy:.1f}"/>'
            f'<circle class="ann-dot" cx="{cx:.1f}" cy="{cy:.1f}" r="6"/>'
            f'<text class="ann-text" x="{text_x:.1f}" y="{cy-2:.1f}" text-anchor="{anchor}">{escape(text)}</text>'
            f'</g>'
        )

    return f'''<svg class="chart" viewBox="0 0 {width} {height}" preserveAspectRatio="none" role="img" aria-label="{escape(y_label)} vs {escape(x_label)} scatter">
<g class="grid">{grid}</g>
{dots}
<g class="axis">{"".join(y_ticks)}</g>
<g>{"".join(x_ticks)}</g>
<text class="label" x="{pad_l+plot_w/2}" y="{height-2}" text-anchor="middle" style="font-weight:600">{escape(x_label)}</text>
<text class="label" x="-{pad_t+plot_h/2}" y="14" text-anchor="middle" transform="rotate(-90)" style="font-weight:600">{escape(y_label)}</text>
{ann_svg}
</svg>'''


def top_list(title: str, ag: str, headers: list[str], rows: list[list[str]]) -> str:
    h_cells = ""
    for i, h in enumerate(headers):
        cls = ' class="num"' if i == len(headers) - 1 else ""
        h_cells += f'<th{cls}>{escape(h)}</th>'
    body = ""
    for i, row in enumerate(rows, 1):
        cells = f'<td>{i:02d}</td>'
        for j, c in enumerate(row):
            if j == 0:
                cells += f'<td class="title">{c}</td>'
            elif j == len(row) - 1:
                cells += f'<td class="num">{c}</td>'
            else:
                cells += f'<td>{c}</td>'
        body += f'<tr>{cells}</tr>'
    return f"""<div class="toplist">
  <h3>{escape(title)}</h3>
  <div class="ag">{escape(ag)}</div>
  <table>
    <thead><tr><th></th>{h_cells}</tr></thead>
    <tbody>{body}</tbody>
  </table>
</div>"""


def footer(sources: list[tuple[str, str]]) -> str:
    src_html = " · ".join(f'<a href="{escape(href)}">{escape(label)}</a>' for label, href in sources)
    return f'<div class="foot">Sources · {src_html}</div>'

_SENTINEL_PAT = re.compile(r'<!-- BEGIN auto:body -->.*?<!-- END auto:body -->', re.S)


def update_dashboard_section(dashboard_path: Path, body: str) -> None:
    """Extract the auto portion of `body` (verdict → top_list) and inject it
    between BEGIN/END auto:body sentinels in `dashboard_path`. The
    hand-curated head, eval-badge, qa-line, Section B (where present), and
    footer extras (e.g. 'Last updated') are preserved.
    """
    try:
        start = body.index('<div class="verdict">')
        end = body.index('<div class="foot">')
    except ValueError as exc:
        raise RuntimeError(
            f"Could not locate verdict/footer markers in body for {dashboard_path}"
        ) from exc
    fragment = body[start:end].rstrip()
    txt = dashboard_path.read_text()
    new_block = f'<!-- BEGIN auto:body -->\n{fragment}\n<!-- END auto:body -->'
    new_txt, n = _SENTINEL_PAT.subn(lambda m: new_block, txt, count=1)
    if n != 1:
        raise RuntimeError(
            f"<!-- BEGIN auto:body --> sentinel not found in {dashboard_path}"
        )
    dashboard_path.write_text(new_txt)
    print(f"  ✓ {dashboard_path.relative_to(BASE)}")

def load_sprout_subset(network: str) -> pd.DataFrame:
    df = _load_sprout_csv()
    df["Date"] = pd.to_datetime(df["Date"], format="%m-%d-%Y", errors="coerce")
    for c in ["Audience", "Net Audience Growth", "Published Posts (Total)",
              "Impressions", "Engagements", "Video Views"]:
        if df[c].dtype == object:
            df[c] = df[c].astype(str).str.replace(",", "").replace({"": None, "nan": None})
        df[c] = pd.to_numeric(df[c], errors="coerce")
    return df[df["Network"] == network].sort_values("Date").reset_index(drop=True)


def fmt(n, kind="int"):
    if pd.isna(n) or n is None:
        return "—"
    if kind == "int":
        return f"{int(n):,}"
    if kind == "pct":
        return f"{n*100:.1f}%" if n < 1 else f"{n:.2f}%"
    if kind == "k":
        return f"{n/1000:.1f}K" if n >= 1000 else f"{int(n):,}"
    return str(n)

def build_overview():
    """Cross-channel rollup."""
    sprout = _load_sprout_csv()
    sprout["Date"] = pd.to_datetime(sprout["Date"], format="%m-%d-%Y", errors="coerce")
    for c in ["Audience", "Net Audience Growth", "Impressions", "Engagements", "Published Posts (Total)"]:
        if sprout[c].dtype == object:
            sprout[c] = sprout[c].astype(str).str.replace(",", "").replace({"": None, "nan": None})
        sprout[c] = pd.to_numeric(sprout[c], errors="coerce")

    last_date = sprout["Date"].max()
    networks = ["Instagram", "LinkedIn", "X", "YouTube"]

    def window_totals(days: int = None):
        """Sum impressions/engagements across all networks for the window."""
        if days is None:
            sub = sprout[sprout["Network"].isin(networks)]
        else:
            sub = sprout[(sprout["Network"].isin(networks)) & (sprout["Date"] >= last_date - pd.Timedelta(days=days))]
        return int(sub["Impressions"].sum()), int(sub["Engagements"].sum())

    impr_7d, eng_7d = window_totals(7)
    impr_30d, eng_30d = window_totals(30)
    impr_90d, eng_90d = window_totals(90)
    impr_all, eng_all = window_totals(None)

    snapshots = []
    for net in networks:
        sub = sprout[sprout["Network"] == net]
        aud = sub.dropna(subset=["Audience"]).sort_values("Date")
        latest = int(aud["Audience"].iloc[-1]) if not aud.empty else 0
        last30 = sub[sub["Date"] >= last_date - pd.Timedelta(days=30)]
        impr = int(last30["Impressions"].sum())
        eng = int(last30["Engagements"].sum())
        snapshots.append({"net": net, "audience": latest, "impr": impr, "eng": eng})

    total_audience = sum(s["audience"] for s in snapshots)
    total_30d_impr = impr_30d
    total_30d_eng = eng_30d
