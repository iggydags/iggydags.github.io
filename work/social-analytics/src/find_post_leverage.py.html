"""Post-level leverage analysis from Sprout's Post Performance export.

Surfaces format concentration, Pareto curves, sender attribution, and per-format yield —
the things you can't compute from daily aggregate data.

Outputs structured findings to stdout. Pipe to a markdown doc when satisfied.
"""
from __future__ import annotations
import warnings
from pathlib import Path
import pandas as pd

warnings.filterwarnings("ignore")
BASE = Path("/Users/dags/Desktop/PubKey/PubKey-Ops/Social")


def _to_num(s: pd.Series) -> pd.Series:
    if s.dtype == object:
        s = s.astype(str).str.replace(",", "", regex=False).replace({"": None, "nan": None, " ": None})
    return pd.to_numeric(s, errors="coerce")


def load_posts() -> pd.DataFrame:
    files = sorted((BASE / "ref/data/overview/source").glob("Post Performance*.csv"))
    if not files:
        raise FileNotFoundError("No Post Performance*.csv in overview/source/")
    df = pd.concat([pd.read_csv(f) for f in files], ignore_index=True)
    df["Date"] = pd.to_datetime(df["Date"], errors="coerce")
    for c in ["Impressions", "Organic Impressions", "Paid Impressions",
              "Reach", "Engagements", "Engagement Rate (per Impression)",
              "Reactions", "Likes", "Comments", "Shares", "Saves",
              "Post Link Clicks", "Video Views", "Subscribers Gained from Video",
              "Reels Skip Rate", "Estimated Minutes Watched"]:
        if c in df.columns:
            df[c] = _to_num(df[c])
    df = df.drop_duplicates(subset=["Post ID"], keep="last").reset_index(drop=True)
    return df


def format_concentration(df: pd.DataFrame, network: str) -> pd.DataFrame:
    sub = df[df["Network"] == network].copy()
    if sub.empty:
        return pd.DataFrame()
    g = sub.groupby("Post Type").agg(
        posts=("Post ID", "count"),
        impressions=("Impressions", "sum"),
        engagements=("Engagements", "sum"),
        eng_per_post=("Engagements", "mean"),
        median_eng=("Engagements", "median"),
        median_impr=("Impressions", "median"),
        median_er=("Engagement Rate (per Impression)", "median"),
    ).round(2)
    total_eng = g["engagements"].sum() or 1
    total_posts = g["posts"].sum() or 1
    g["effort_share_%"] = (g["posts"] / total_posts * 100).round(1)
    g["eng_share_%"] = (g["engagements"] / total_eng * 100).round(1)
    g["leverage"] = (g["eng_share_%"] / g["effort_share_%"]).round(2)
    return g.sort_values("eng_share_%", ascending=False)


def pareto(df: pd.DataFrame, network: str, metric: str = "Engagements") -> dict:
    sub = df[df["Network"] == network].copy()
    sub = sub[sub[metric].notna() & (sub[metric] > 0)]
    if sub.empty:
        return {}
    sub = sub.sort_values(metric, ascending=False).reset_index(drop=True)
    total = sub[metric].sum()
    out = {}
    for pct in [5, 10, 20, 50]:
        n = max(1, int(len(sub) * pct / 100))
        share = sub.head(n)[metric].sum() / total * 100
        out[f"Top {pct}% (n={n})"] = f"{share:.1f}% of {metric.lower()}"
    out["Zero-engagement posts"] = f"{int((df[df['Network'] == network][metric] == 0).sum())} of {int((df['Network'] == network).sum())} ({(df[df['Network'] == network][metric] == 0).mean() * 100:.1f}%)"
    return out


def top_posts(df: pd.DataFrame, network: str, n: int = 10) -> pd.DataFrame:
    sub = df[df["Network"] == network].copy()
    sub = sub[sub["Engagements"] > 0].sort_values("Engagements", ascending=False).head(n)
    cols = ["Date", "Post Type", "Engagements", "Impressions",
            "Engagement Rate (per Impression)", "Post"]
    cols = [c for c in cols if c in sub.columns]
    out = sub[cols].copy()
    if "Post" in out.columns:
        out["Post"] = out["Post"].astype(str).str.slice(0, 80) + "..."
    if "Date" in out.columns:
        out["Date"] = out["Date"].dt.strftime("%Y-%m-%d")
    return out


def sender_attribution(df: pd.DataFrame, network: str) -> pd.DataFrame:
    sub = df[df["Network"] == network].copy()
    if "Sent by" not in sub.columns or sub.empty:
        return pd.DataFrame()
    sub["Sent by"] = sub["Sent by"].fillna("(unattributed)").astype(str).str.strip().replace("", "(unattributed)")
    g = sub.groupby("Sent by").agg(
        posts=("Post ID", "count"),
        avg_eng=("Engagements", "mean"),
        avg_impr=("Impressions", "mean"),
        med_er=("Engagement Rate (per Impression)", "median"),
        total_eng=("Engagements", "sum"),
    ).round(1)
    g["eng_per_post"] = g["avg_eng"].round(1)
    g = g.sort_values("posts", ascending=False)
    return g


def paid_vs_organic(df: pd.DataFrame, network: str) -> dict:
    sub = df[df["Network"] == network].copy()
    if sub.empty or "Paid Impressions" not in sub.columns:
        return {}
    sub["Paid Impressions"] = sub["Paid Impressions"].fillna(0)
    sub["has_paid"] = sub["Paid Impressions"] > 0
    paid = sub[sub["has_paid"]]
    org = sub[~sub["has_paid"]]
    if paid.empty:
        return {"paid_post_count": 0, "note": "no paid posts in window"}
    return {
        "paid_posts": len(paid),
        "organic_posts": len(org),
        "paid_avg_impressions": int(paid["Impressions"].mean()) if not paid.empty else 0,
        "organic_avg_impressions": int(org["Impressions"].mean()) if not org.empty else 0,
        "paid_avg_engagements": int(paid["Engagements"].mean()) if not paid.empty else 0,
        "organic_avg_engagements": int(org["Engagements"].mean()) if not org.empty else 0,
        "paid_subsidy_factor": round(paid["Impressions"].mean() / org["Impressions"].mean(), 1) if not org.empty and org["Impressions"].mean() else None,
    }


def main():
    df = load_posts()
    print("=" * 80)
    print("PUBKEY POST PERFORMANCE · LEVERAGE FINDINGS · 2026-05-15")
    print("=" * 80)
    print(f"Total posts: {len(df):,} · Networks: {sorted(df['Network'].dropna().unique())}")
    print(f"Window: {df.Date.min().date()} → {df.Date.max().date()}")
    print()

    print("─" * 80)
    print("POSTS PER NETWORK")
    print("─" * 80)
    print(df.groupby("Network").size().sort_values(ascending=False).to_string())
    print()

    for network in ["Instagram", "LinkedIn", "X", "Facebook", "TikTok"]:
        if not (df["Network"] == network).any():
            continue
        print("█" * 80)
        print(f"  {network.upper()}")
        print("█" * 80)

        print(f"\n--- Format concentration ({network}) ---")
        fc = format_concentration(df, network)
        if not fc.empty:
            print(fc.to_string())

        print(f"\n--- Pareto curve · engagements ({network}) ---")
        for k, v in pareto(df, network, "Engagements").items():
            print(f"  {k:<28} {v}")

        if "Video Views" in df.columns and (df[df["Network"] == network]["Video Views"] > 0).any():
            print(f"\n--- Pareto curve · video views ({network}) ---")
            for k, v in pareto(df, network, "Video Views").items():
                print(f"  {k:<28} {v}")

        print(f"\n--- Sender attribution ({network}) ---")
        sa = sender_attribution(df, network)
        if not sa.empty:
            print(sa.head(8).to_string())

        print(f"\n--- Paid vs organic ({network}) ---")
        pvo = paid_vs_organic(df, network)
        for k, v in pvo.items():
            print(f"  {k:<28} {v}")

        print(f"\n--- Top 5 posts ({network}) ---")
        tp = top_posts(df, network, n=5)
        if not tp.empty:
            print(tp.to_string(index=False))
        print()


if __name__ == "__main__":
    main()
