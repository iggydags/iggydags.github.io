def _read_concat(folder: Path, glob_pattern: str, date_col: str = "Date", date_format: str = "%m-%d-%Y") -> pd.DataFrame:
    """Read all CSVs matching glob in folder, concat, dedupe by date_col, sort."""
    files = sorted(folder.glob(glob_pattern))
    if not files:
        raise FileNotFoundError(f"No files matching {glob_pattern} in {folder}")
    dfs = [pd.read_csv(f) for f in files]
    df = pd.concat(dfs, ignore_index=True)
    df[date_col] = pd.to_datetime(df[date_col], format=date_format, errors="coerce")
    df = df.drop_duplicates(subset=[date_col], keep="last")
    return df.sort_values(date_col).reset_index(drop=True)

def load_sprout():
    """Return Sprout daily data, normalized. Globs all Profile Performance CSVs and dedupes by (Date, Network, Profile)."""
    files = sorted(SPROUT_FOLDER.glob(SPROUT_GLOB))
    if not files:
        raise FileNotFoundError(f"No files matching {SPROUT_GLOB} in {SPROUT_FOLDER}")
    df = pd.concat([pd.read_csv(f) for f in files], ignore_index=True)
    df["Date"] = pd.to_datetime(df["Date"], format="%m-%d-%Y", errors="coerce")
    dedupe_keys = [c for c in ["Date", "Network", "Profile"] if c in df.columns]
    df = df.drop_duplicates(subset=dedupe_keys, keep="last")
    # Coerce numeric columns
    numeric_cols = [
        "Audience",
        "Net Audience Growth",
        "Audience Gained",
        "Published Posts (Total)",
        "Impressions",
        "Video Views",
        "Engagements",
        "Reactions",
        "Comments",
        "Shares",
        "Post Link Clicks",
        "Saves",
    ]
    for c in numeric_cols:
        # Sprout exports comma-formatted strings ("1,643") for some columns — strip commas first
        if df[c].dtype == object:
            df[c] = df[c].astype(str).str.replace(",", "", regex=False).str.strip()
            df[c] = df[c].replace({"": None, "nan": None})
        df[c] = pd.to_numeric(df[c], errors="coerce")
    # Engagement rate: strip % and convert to fraction
    er = df["Engagement Rate (per Impression)"].astype(str).str.rstrip("%")
    df["Engagement Rate"] = pd.to_numeric(er, errors="coerce") / 100
    df = df.sort_values("Date")
    return df

def add_sprout_daily_tab(wb, network, profile=None):
    """Add a Daily tab with Sprout timeseries for the given network/profile."""
    sprout = load_sprout()
    sub = sprout[sprout["Network"] == network].copy()
    if profile:
        sub = sub[sub["Profile"] == profile]
    sub = sub[SPROUT_DAILY_COLS]
    ws = wb.create_sheet("Daily")
    add_title(ws, f"{network} — Daily Timeseries (Sprout)", subtitle=f"Source: ref/data/overview/source/Profile Performance ...csv · {sub.Date.min().date() if not sub.empty else 'no data'} → {sub.Date.max().date() if not sub.empty else 'no data'}")
    for i, h in enumerate(SPROUT_DAILY_HEADERS, 1):
        ws.cell(row=6, column=i, value=h)
    style_header_row(ws, 6, len(SPROUT_DAILY_HEADERS))
    for r_idx, row in enumerate(sub.itertuples(index=False), start=7):
        for c_idx, val in enumerate(row, start=1):
            cell = ws.cell(row=r_idx, column=c_idx)
            if c_idx == 1:
                cell.value = val.date() if pd.notna(val) else None
                cell.number_format = "yyyy-mm-dd"
            elif c_idx == 8:  # Eng Rate
                cell.value = float(val) if pd.notna(val) else None
                cell.number_format = "0.00%"
            else:
                cell.value = int(val) if pd.notna(val) and val == int(val) else (float(val) if pd.notna(val) else None)
    last_row = 6 + len(sub)
    apply_body_font(ws, 7, last_row, 1, len(SPROUT_DAILY_HEADERS))
    set_column_widths(ws, [13, 11, 11, 8, 13, 13, 13, 11, 11, 11, 9, 11, 9])
    ws.freeze_panes = "B7"
    return last_row

def style_header_row(ws, row, last_col):
    for col in range(1, last_col + 1):
        c = ws.cell(row=row, column=col)
        c.font = HEADER_FONT
        c.fill = HEADER_FILL
        c.alignment = Alignment(horizontal="left", vertical="center")
        c.border = BORDER


def apply_body_font(ws, start_row, end_row, start_col, end_col):

        for c in range(start_col, end_col + 1):
            cell = ws.cell(row=r, column=c)
            if cell.font.name != FONT_NAME:
